#Stressed character: Command a unit
character_event = {
	id = 65000
	desc = EVTDESC65000
	picture = GFX_evt_crusaders
	
	only_men = yes
	capable_only = yes
	prisoner = no
	
	is_triggered_only = yes

	trigger = {
		war = yes
		in_command = yes
		OR = {
			trait = stressed
			trait = stressed_hidden
		}
		NOT = {
			has_character_modifier = stressed_commander
			has_character_modifier = misplaced_food
		}
	}
	
	weight_multiplier = {
		factor = 1
	}
	
	option = {
		name = EVTOPTA65000
		add_character_modifier = {
			name = stressed_commander
			duration = 365
		}
	}
	option = {
		name = EVTOPTB65000
		troops = -0.2
	}
}

### Ruler lashes out in anger to hide internal frustrations ###
character_event = {
	id = 65011
	desc = EVTDESC65011
	picture = GFX_evt_scandal
	
	only_men = yes
	capable_only = yes
	prisoner = no
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			trait = stressed
			trait = stressed_hidden
		}
		NOT = {
			OR = {
				trait = kind
				trait = kind_hidden
			}
		}
		OR = {
			spouse = {
				is_abroad = no
				prisoner = no
				is_alive = yes
				NOT = {
					OR = {
						trait = stressed
						trait = stressed_hidden
					}
				}
			}
			any_child = {
				is_abroad = no
				prisoner = no
				is_primary_heir = yes
			}
			any_courtier = {
				is_abroad = no
				prisoner = no
				is_councillor = yes
			}
			any_courtier = {
				is_abroad = no
				prisoner = no
				is_councillor = no
			}
		}
	}
	
	weight_multiplier = {
		factor = 1
	
		modifier = {
			factor = 2
			OR = {
				trait = arbitrary
				trait = arbitrary_hidden
			}
		}
		modifier = {
			factor = 2
			OR = {
				trait = cruel
				trait = cruel_hidden
			}
		}
	}
	
	option = {
		name = EVTOPTA65011
		trigger = {
			is_married = yes
		}
		spouse = {
			character_event = {
				id = 65012
				tooltip = EVTTOOLTIPA65012
			}
		}
	}
	option = {
		name = EVTOPTB65011
		trigger = {
			any_child = {
				is_primary_heir = yes
			}
		}
		random_child = {
			limit = {
				is_primary_heir = yes
			}
			character_event = {
				id = 65012
				tooltip = EVTTOOLTIPB65012
			}
		}
	}
	
	option = {
		name = EVTOPTC65011

		}
		random_courtier = {
			limit = {
				is_councillor = yes
			}
			character_event = {
				id = 65012
				tooltip = EVTTOOLTIPC65012
			}
		}
	}
	option = {
		name = EVTOPTD65011
		trigger = {
			any_courtier = {
				is_councillor = no
			}
		}
		random_courtier = {
			limit = {
				is_councillor = no
			}

				id = 65012
				tooltip = EVTTOOLTIPD65012

#Cruel character educates child
character_event = {
	id = 65020
	desc = EVTDESC65020
	picture = GFX_evt_guardian
	
	capable_only = yes
	prisoner = no
	
	is_triggered_only = yes
	
	trigger = {
		any_courtier = {
			guardian = {
				character = ROOT
			}
			OR = {
				trait = weak
				trait = weak_hidden
			}
			age = 9
			NOT = {
				age = 16
			}
			NOT = {
				OR = {
					trait = brave
					trait = brave_hidden
				}
				OR = {
					trait = maimed
					trait = maimed_hidden
				}
			}
		}
	}
	
	weight_multiplier = {
		factor = 1
	}
	
	option = {
		name = EVTOPTA65020 #Try to beat some sense in the child
		if = {
			limit = { religion = jain }
			piety = -20
			custom_tooltip = { text = jain_ahimsa_tooltip }
		}
		random_child = {
			limit = {
				guardian = {
					character = ROOT
				}
				OR = {
					trait = weak
					trait = weak_hidden
				}
				age = 9
				NOT = {
					age = 15
				}
				NOT = {
					OR = {
						trait = brave
						trait = brave_hidden
					}
					OR = {
						trait = maimed
						trait = maimed_hidden
					}
				}
			}
			opinion = {
				modifier = opinion_cruel
				who = ROOT
				years = 5
			}
			random = {
				chance = 50
				add_trait = maimed_hidden
