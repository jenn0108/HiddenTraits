character_event = {
	id = 36000
	desc = "EVTDESC36000"
	picture = "GFX_evt_hunt"

	min_age =  16
	only_men = yes
	capable_only = yes
	prisoner = no

	trigger = {
		has_character_modifier = holding_grand_hunt
		NOT = {	health_traits = 1 }
		OR = {
			NOT = { has_character_flag = hunt_event_happened }
			had_character_flag = {
				flag = hunt_event_happened
				days = 3
			}
		}
		had_character_flag = {
			flag = hunt_begin
			days = 3
		}		
	}
	
	immediate = {
		set_character_flag = hunt_event_happened
	}

	mean_time_to_happen = {
		days = 200
		
		modifier = {
			factor = 0.33
			NOT = { has_character_flag = hunt_event_happened }
		}	
		
		modifier = {
			factor = 1.5
			has_character_flag = hunt_event_happened
		}
		
		modifier = {
			factor = 1.5
			OR = {
				trait = content
				trait = content_hidden
			}
		}
		modifier = {
			factor = 0.8
			OR = {
				trait = hunter
				trait = hunter_hidden
			}
		}
		modifier = {
			factor = 5.0
			martial = 15
		}
		modifier = {
			factor = 2.0
			OR = {
				trait = slothful
				trait = slothful_hidden
			}
		}
		modifier = {
			factor = 0.5
			has_character_flag = hunt_good_hounds
		}
		modifier = {
			factor = 1.25
			OR = {
				trait = craven
				trait = craven_hidden
			}
		}

	}

	option = {
		name = "EVTOPTA36000"
		trigger = {
			NOT = {
				OR = {
					trait = craven

				}
			}
		}
		prestige = 5
	}

	option = {
		name = "EVTOPTB36000"
		tooltip_info = craven
		trigger = {
			OR = {
				trait = craven
				trait = craven_hidden
			}
		}
		prestige = -10
	}
	option = {
		name = "EVTOPTC36000"
		trigger = {
			NOT = {
				 OR = {
					trait = greedy
					trait = greedy_hidden
				} 
			}
